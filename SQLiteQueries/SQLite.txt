PRAGMA foreign_keys = ON;

 CREATE TABLE events(
   event_id INT PRIMARY KEY     NOT NULL,
   event_name     CHARACTER    NOT NULL UNIQUE,
   ticket_price   NUMERIC 	  NOT NULL,
   event_date				DATE	NOT NULL,
   event_location       INT     NOT NULL,
   event_type       INT     NOT NULL,
   FOREIGN KEY(event_location) REFERENCES locations(location_id),
   FOREIGN KEY(event_type) REFERENCES types(type_id)
);

CREATE TABLE users(
   user_id INT PRIMARY KEY     NOT NULL,
   user_name	VARCHAR    NOT NULL UNIQUE,
   user_email     VARCHAR    NOT NULL UNIQUE,
   password   CHARACTER	  NOT NULL,
   is_admin       INT     NOT NULL
);

CREATE TABLE events_users(
   ev_id     INT    NOT NULL,
   u_id		INT    NOT NULL,
   FOREIGN KEY(ev_id) REFERENCES events(event_id),
   FOREIGN KEY(u_id) REFERENCES users(user_id)
);

CREATE TABLE locations(
   location_id INT PRIMARY KEY     NOT NULL,
   location_name		CHARACTER    NOT NULL  UNIQUE,
);

CREATE TABLE types(
   type_id INT PRIMARY KEY     NOT NULL,
   type_name	CHARACTER    NOT NULL  UNIQUE,
);
